**_App.vue_**

```vue
<template>
  <div class="app">
    <div class="tabs">
      <template v-for="(item, index) in tabs" :key="item">
        <button
          :class="{ active: currentTab === item }"
          @click="itemClick(item)"
        >
          {{ item }}
        </button>
      </template>
    </div>
    <div class="view">
      <!-- 1. 第一种做法，里面有大量的逻辑判断 v-if -->
      <!-- 
      <template v-if="currentIndex ===0">
      <home></home>
      </template>
      <template v-else-if="currentIndex ===1">
        <about></about>
      </template>
      <template v-else-if="currentIndex ===2">
        <category></category>
      </template> 
    -->
      <!-- 2.方式二：动态组件的实现 is后面是组件名字，不区分大小写
      is后面是哪个组件名就显示哪个组件，
      is中的组件需要来自两个地方
      1. 全局注册的组件，2.局部注册的组件-->
      <!-- <component is="about"></component> -->
      <!-- 动态绑定属性 v-bind   : -->
      <!-- 
      一：动态组件传值 
      二：监听组件内部事件
      -->
      <!-- 动态组件每一次切换都会创建和销毁 vue2中created  destroyed
      但是会多两个钩子函数 activated 和 deactivated 如果切换到当前组件就是 activated 切走就是 deactivated-->
      <component
        name="zql"
        :age="18"
        @homeClick="homeClick"
        :is="currentTab"
      ></component>
    </div>
  </div>
</template>

<script>
import About from "./views/About.vue";
import Category from "./views/Category.vue";
import Home from "./views/Home.vue";
export default {
  components: {
    About,
    Category,
    Home,
  },
  data() {
    return {
      tabs: ["home", "about", "category"],
      currentTab: "Home",
    };
  },
  methods: {
    itemClick(item) {
      this.currentTab = item;
    },
    // 三：动态组件事件监听 payload负载负荷，火箭里面携带的燃料 负载
    homeClick(payload) {
      console.log("homeClick", payload);
    },
  },
};
</script>
<!-- scope样式只在当前组件上面有效 -->
<style scoped>
.active {
  color: red;
  font-weight: 700;
}
</style>
```

#### keep-alive

![image-20221206190328042](http://www.zhangqilong.cn/img/qlBlog_images/Vue%E5%9F%BA%E7%A1%80/18_%E5%8A%A8%E6%80%81%E7%BB%84%E4%BB%B6%E7%9A%84%E4%BD%BF%E7%94%A8.assets/image-20221206190328042.png)

```vue
<template>
  <div class="app">
    <div class="tabs">
      <template v-for="(item, index) in tabs" :key="item">
        <button
          :class="{ active: currentTab === item }"
          @click="itemClick(item)"
        >
          {{ item }}
        </button>
      </template>
    </div>
    <div class="view">
      <!-- 4. 那么其实很简单，我们只需要在组件外面套一个 
        <keep-alive></keep-alive> 就可以帮助我们缓存起来了，不会被销毁 
      -->
      <!-- 5.include:组件的名称来自于组件定义时候的选项 匹配首先检查组件自身的 name选项
      如果我们不在组件中进行写name的值，那么还是没有缓存的 6.见home-->
      <!-- 7. exclude 里面的东西都不会被缓存起来 
      8. max 最多可以缓存多少组件实例-->
      <keep-alive include="home" exclude="about" max="1">
        <component :is="currentTab"></component>
      </keep-alive>
    </div>
  </div>
</template>
<!-- 
1. 思考一个问题，就是当我们组件来来回切换的时候，如果从Home组件切换到about或者其他组件
那也就意味着 Home组件就被销毁掉了 我们可以去Home组件里测试生命周期
测试结果就是 Home组件会被来回的创建和销毁 性能不高，那么我们希望Home组件 keep alive
-->
<script>
import About from "./views/About.vue";
import Category from "./views/Category.vue";
import Home from "./views/Home.vue";
export default {
  components: {
    About,
    Category,
    Home,
  },
  data() {
    return {
      tabs: ["home", "about", "category"],
      currentTab: "Home",
    };
  },
  methods: {
    itemClick(item) {
      this.currentTab = item;
    },
    // 三：动态组件事件监听 payload负载负荷，火箭里面携带的燃料 负载
    homeClick(payload) {
      console.log("homeClick", payload);
    },
  },
};
</script>
<!-- scope样式只在当前组件上面有效 -->
<style scoped>
.active {
  color: red;
  font-weight: 700;
}
</style>
```
